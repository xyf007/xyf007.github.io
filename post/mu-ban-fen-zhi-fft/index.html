【模板】分治 FFT

· 2020-12-21 · # 模板

给出 g1,g2,…,gng_1,g_2,\ldots,g_ng1​,g2​,…,gn​，求解 fi=∑j=1ifi−jgjf_i=\sum_{j=1}^if_{i-j}g_jfi​=∑j=1i​fi−j​gj​。

看起来就是 FFT/NTT，但是后面的值依赖于前面的值，于是像 CDQ 分治一样，算前面的值对后面的贡献。

时间复杂度 O(nlog⁡2n)O(n \log^2 n)O(nlog2n)

#include <iostream>
#include <iomanip>
#include <cstdio>
#include <algorithm>
#include <cstring>
constexpr int kMod = 998244353, kG = 3;
int n, a[400001], b[400001], rev[400001], lg[400001], g[400001], ig[400001], A[400001], B[400001];
int Ksm(int a, int b)
{
    int s = 1;
    while (b)
    {
        if (b & 1)
        {
            s = s * static_cast<long long>(a) % kMod;
        }
        a = a * static_cast<long long>(a) % kMod;
        b >>= 1;
    }
    return s;
}
void NTT(int *a, int n, int op)
{
    for (int i = 0; i < n; i++)
    {
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (lg[n] - 1));
    }
    for (int i = 0; i < n; i++)
    {
        if (i < rev[i])
        {
            std::swap(a[i], a[rev[i]]);
        }
    }
    for (int mid = 1; mid < n; mid <<= 1)
    {
        int wn = op == 1 ? g[mid << 1] : ig[mid << 1], a1, a2;
        for (int i = 0; i < n; i += mid << 1)
        {
            int w = 1;
            for (int j = 0; j < mid; j++)
            {
                a1 = a[i + j];
                a2 = static_cast<long long>(w) * a[i + j + mid] % kMod;
                a[i + j] = (a1 + a2) % kMod;
                a[i + j + mid] = (a1 - a2 + kMod) % kMod;
                w = w * static_cast<long long>(wn) % kMod;
            }
        }
    }
    if (op == -1)
    {
        int inv = Ksm(n, kMod - 2);
        for (int i = 0; i < n; i++)
        {
            a[i] = a[i] * static_cast<long long>(inv) % kMod;
        }
    }
}
void Solve(int l, int r)
{
    if (l + 1 >= r)
    {
        return;
    }
    int mid = (l + r) >> 1;
    Solve(l, mid);
    memcpy(A, a + l, 4 * (mid - l));
    memset(A + mid - l, 0, 4 * (r - mid));
    memcpy(B, b, 4 * (r - l));
    NTT(A, r - l, 1);
    NTT(B, r - l, 1);
    for (int i = 0; i < r - l; i++)
    {
        A[i] = A[i] * static_cast<long long>(B[i]) % kMod;
    }
    NTT(A, r - l, -1);
    for (int i = mid; i < r; i++)
    {
        a[i] = (a[i] + A[i - l]) % kMod;
    }
    Solve(mid, r);
}
int main(int argc, char const *argv[])
{
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    a[0] = 1;
    std::cin >> n;
    for (int i = 1; i < n; i++)
    {
        std::cin >> b[i];
    }
    int len = 1, cnt = 0;
    while (len < n)
    {
        cnt++;
        len <<= 1;
    }
    g[len] = Ksm(kG, (kMod - 1) >> cnt);
    ig[len] = Ksm(g[len], kMod - 2);
    for (int i = cnt - 1; i >= 0; i--)
    {
        g[1 << i] = g[1 << (i + 1)] * static_cast<long long>(g[1 << (i + 1)]) % kMod;
        ig[1 << i] = ig[1 << (i + 1)] * static_cast<long long>(ig[1 << (i + 1)]) % kMod;
    }
    for (int i = 1; i <= cnt; i++)
    {
        lg[1 << i] = i;
    }
    Solve(0, len);
    for (int i = 0; i < n; i++)
    {
        std::cout << a[i] << ' ';
    }
    return 0;
}
