注意计算的时候存在相互递归，因此三个要一起算，用结构体存答案。

Node Solve(long long a, long long b, long long c, long long n) {
  Node ans;
  long long bc = b / c, c2 = mul(n, n + 1, kInv2);
  if (a == 0) {
    ans.f = mul(n + 1, bc);
    ans.g = mul(c2, bc);
    ans.h = mul(n + 1, bc, bc);
    return ans;
  }
  long long ac = a / c, c3 = mul(n, n + 1, 2 * n + 1, kInv6);
  if (a >= c || b >= c) {
    Node r = Solve(a % c, b % c, c, n);
    ans.f = add(mul(c2, ac), mul(n + 1, bc), r.f);
    ans.g = add(mul(c3, ac), mul(c2, bc), r.g);
    ans.h = add(mul(c3, ac, ac), mul(n + 1, bc, bc), mul(n, n + 1, ac, bc),
                mul(2 * bc, r.f), mul(2 * ac, r.g), r.h);
    return ans;
  }
  long long m = (a * n + b) / c;
  Node r = Solve(c, c - b - 1, a, m - 1);
  ans.f = sub(mul(n, m), r.f);
  ans.g = sub(mul(c2, m), mul(r.f, kInv2), mul(r.h, kInv2));
  ans.h = sub(mul(n, m, m + 1), r.g, r.g, r.f, r.f, ans.f);
  return ans;
}
