【模板】杜教筛

· 2020-08-18 · # 模板

杜教筛可以在亚线性时间内筛积性函数前缀和。

杜教筛的主要思想就是构造一个 S(n)S(n)S(n) 关于 S(⌊ni⌋)S(\left\lfloor\frac{n}{i}\right\rfloor)S(⌊in​⌋) 的递推式。因为对于一个数论函数 ggg，必然满足

∑i=1n∑d∣if(d)g(id)=∑d=1n∑i=1⌊ni⌋g(d)f(i)=∑i=1ng(i)∑j=1⌊ni⌋f(j)=∑i=1ng(i)S(⌊ni⌋)\begin{aligned} \sum\limits_{i=1}^n\sum\limits_{d \mid i}f(d)g\left(\dfrac{i}{d}\right)&=\sum\limits_{d=1}^n\sum\limits_{i=1}^{\lfloor\frac{n}{i}\rfloor}g(d)f(i)\\ &=\sum\limits_{i=1}^ng(i)\sum\limits_{j=1}^{\lfloor\frac{n}{i}\rfloor}f(j)\\ &=\sum\limits_{i=1}^ng(i)S\left(\left\lfloor\dfrac{n}{i}\right\rfloor\right) \end{aligned} i=1∑n​d∣i∑​f(d)g(di​)​=d=1∑n​i=1∑⌊in​⌋​g(d)f(i)=i=1∑n​g(i)j=1∑⌊in​⌋​f(j)=i=1∑n​g(i)S(⌊in​⌋)​

即

∑i=1n(f∗g)(i)=∑i=1ng(i)S(⌊ni⌋)\sum\limits_{i=1}^n(f*g)(i)=\sum\limits_{i=1}^ng(i)S\left(\left\lfloor\dfrac{n}{i}\right\rfloor\right) i=1∑n​(f∗g)(i)=i=1∑n​g(i)S(⌊in​⌋)

那么可以得到递推式

g(1)S(n)=∑i=1n(f∗g)(i)−∑i=2ng(i)S(⌊ni⌋)g(1)S(n)=\sum\limits_{i=1}^n(f*g)(i)-\sum\limits_{i=2}^ng(i)S\left(\left\lfloor\dfrac{n}{i}\right\rfloor\right) g(1)S(n)=i=1∑n​(f∗g)(i)−i=2∑n​g(i)S(⌊in​⌋)

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <unordered_map>
using namespace std;
const int N = 1664512;
int t, n, not_prime[N], p[150001], tot, mu[N];
long long phi[N];
unordered_map<int, int> summu;
unordered_map<int, long long> sumphi;
void init()
{
    not_prime[1] = mu[1] = phi[1] = 1;
    for (int i = 2; i < N; i++)
    {
        if (!not_prime[i])
        {
            p[++tot] = i;
            mu[i] = -1;
            phi[i] = i - 1;
        }
        for (int j = 1; j <= tot && i * p[j] < N; j++)
        {
            not_prime[i * p[j]] = 1;
            if (!(i % p[j]))
            {
                mu[i * p[j]] = 0;
                phi[i * p[j]] = phi[i] * p[j];
                break;
            }
            mu[i * p[j]] = -mu[i];
            phi[i * p[j]] = phi[i] * (p[j] - 1);
        }
    }
    for (int i = 2; i < N; i++)
    {
        mu[i] += mu[i - 1];
        phi[i] += phi[i - 1];
    }
}
int calc_mu(int x)
{
    if (x < N)
    {
        return mu[x];
    }
    if (summu.count(x))
    {
        return summu[x];
    }
    int r, ans = 0;
    for (unsigned int l = 2; l <= x; l = r + 1)
    {
        r = x / (x / l);
        ans += (r - l + 1) * calc_mu(x / l);
    }
    return summu[x] = 1 - ans;
}
long long calc_phi(int x)
{
    if (x < N)
    {
        return phi[x];
    }
    if (sumphi.count(x))
    {
        return sumphi[x];
    }
    int r;
    long long ans = 0;
    for (unsigned int l = 2; l <= x; l = r + 1)
    {
        r = x / (x / l);
        ans += (r - l + 1) * calc_phi(x / l);
    }
    return sumphi[x] = 1LL * x * (x + 1LL) / 2 - ans;
}
int main(int argc, char const *argv[])
{
    init();
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d", &n);
        printf("%lld %d\n", calc_phi(n), calc_mu(n));
    }
    return 0;
}
