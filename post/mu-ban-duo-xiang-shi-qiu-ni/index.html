【模板】多项式求逆

· 2020-12-19 · # 模板

当只有一项的时候，显然 [x0]f−1(x)=([x0]f(x))−1\left[x^0\right]f^{-1}(x)=\left(\left[x^0\right]f(x)\right)^{-1}[x0]f−1(x)=([x0]f(x))−1。
现在假设已经求出了 f(x)f(x)f(x) 在模 xn2x^{\frac{n}{2}}x2n​ 意义下的逆元 f0−1(x)f_0^{-1}(x)f0−1​(x)，那么有

f(x)f0−1(x)≡1(modxn2)f(x)f−1(x)≡1(modxn2)∴f−1(x)−f0−1(x)≡0(modxn2)\begin{aligned} f(x)f_0^{-1}(x) &\equiv 1 &\pmod {x^{\frac{n}{2}}}\\ f(x)f^{-1}(x) &\equiv 1 &\pmod {x^{\frac{n}{2}}}\\ \therefore f^{-1}(x)-f_0^{-1}(x) &\equiv 0 &\pmod {x^\frac{n}{2}} \end{aligned} f(x)f0−1​(x)f(x)f−1(x)∴f−1(x)−f0−1​(x)​≡1≡1≡0​(modx2n​)(modx2n​)(modx2n​)​

两边平方，同乘 f(x)f(x)f(x) 并移项可得

f−2(x)−2f−1(x)f0−1(x)+f0−2(x)≡0(modxn)f−1(x)≡f0−1(x)(2−f(x)f0−1(x))(modxn)\begin{aligned} f^{-2}(x)-2f^{-1}(x)f_0^{-1}(x)+f_0^{-2}(x) &\equiv 0 &\pmod {x^n}\\ f^{-1}(x) &\equiv f_0^{-1}(x)(2-f(x)f_0^{-1}(x)) &\pmod {x^n} \end{aligned} f−2(x)−2f−1(x)f0−1​(x)+f0−2​(x)f−1(x)​≡0≡f0−1​(x)(2−f(x)f0−1​(x))​(modxn)(modxn)​

递归计算即可。
时间复杂度：T(n)=T(n2)+O(nlog⁡n)=O(nlog⁡n)T(n)=T\left(\dfrac{n}{2}\right)+O(n \log n)=O(n \log n)T(n)=T(2n​)+O(nlogn)=O(nlogn)。

注意事项：最后一次求逆会倍增到长度的两倍，因此预处理要多一次。

#include <iostream>
#include <iomanip>
#include <cstdio>
#include <algorithm>
#include <cstring>
constexpr int kMod = 998244353, kG = 3;
int n, a[400001], b[400001], rev[400001], g[400001], ig[400001], lg[400001];
int Ksm(int a, int b)
{
    int s = 1;
    while (b)
    {
        if (b & 1)
        {
            s = s * static_cast<long long>(a) % kMod;
        }
        a = a * static_cast<long long>(a) % kMod;
        b >>= 1;
    }
    return s;
}
void NTT(int *a, int n, int op)
{
    for (int i = 0; i < n; i++)
    {
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (lg[n] - 1));
    }
    for (int i = 0; i < n; i++)
    {
        if (i < rev[i])
        {
            std::swap(a[i], a[rev[i]]);
        }
    }
    for (int mid = 1; mid < n; mid <<= 1)
    {
        int wn = op == 1 ? g[mid << 1] : ig[mid << 1], a1, a2;
        for (int i = 0; i < n; i += mid << 1)
        {
            int w = 1;
            for (int j = 0; j < mid; j++)
            {
                a1 = a[i + j];
                a2 = static_cast<long long>(w) * a[i + j + mid] % kMod;
                a[i + j] = (a1 + a2) % kMod;
                a[i + j + mid] = (a1 - a2 + kMod) % kMod;
                w = w * static_cast<long long>(wn) % kMod;
            }
        }
    }
    if (op == -1)
    {
        int inv = Ksm(n, kMod - 2);
        for (int i = 0; i < n; i++)
        {
            a[i] = a[i] * static_cast<long long>(inv) % kMod;
        }
    }
}
void Get_inv(const int *a, int n, int *b)
{
    static int tmp[400001];
    b[0] = Ksm(a[0], kMod - 2);
    for (int len = 2; len <= n; len <<= 1)
    {
        memcpy(tmp, a, 4 * len);
        memset(tmp + len, 0, 4 * len);
        NTT(b, len << 1, 1);
        NTT(tmp, len << 1, 1);
        for (int i = 0; i < (len << 1); i++)
        {
            b[i] = static_cast<long long>(b[i]) * (2 - static_cast<long long>(b[i]) * tmp[i] % kMod + kMod) % kMod;
        }
        NTT(b, len << 1, -1);
        memset(b + len, 0, 4 * len);
    }
}
int main(int argc, char const *argv[])
{
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::cin >> n;
    for (int i = 0; i < n; i++)
    {
        std::cin >> a[i];
    }
    int len = 1, cnt = 0;
    while (len < n)
    {
        cnt++;
        len <<= 1;
    }
    cnt++;
    len <<= 1;
    g[len] = Ksm(kG, (kMod - 1) >> cnt);
    ig[len] = Ksm(g[len], kMod - 2);
    for (int i = cnt - 1; i >= 0; i--)
    {
        g[1 << i] = g[1 << (i + 1)] * static_cast<long long>(g[1 << (i + 1)]) % kMod;
        ig[1 << i] = ig[1 << (i + 1)] * static_cast<long long>(ig[1 << (i + 1)]) % kMod;
    }
    for (int i = 1; i <= cnt; i++)
    {
        lg[1 << i] = i;
    }
    Get_inv(a, len >>= 1, b);
    for (int i = 0; i < n; i++)
    {
        std::cout << b[i] << ' ';
    }
    return 0;
}
