【模板】快速莫比乌斯/沃尔什变换（FMT/FWT）

· 2020-12-18 · # 模板

快速莫比乌斯/沃尔什变换

通常见到的多项式卷积是 Ck=∑i+j=kAiBjC_k=\sum_{i+j=k}A_iB_jCk​=∑i+j=k​Ai​Bj​ 形式的，可以用 FFT/NTT 做。
如果将操作符换成集合运算符，比如如下形式

Ck=∑i∣j=kAiBjCk=∑i&j=kAiBjCk=∑i⊕j=kAiBj\begin{aligned} C_k&=\sum_{i|j=k}A_iB_j\\ C_k&=\sum_{i \& j=k}A_iB_j\\ C_k&=\sum_{i \oplus j=k}A_iB_j \end{aligned} Ck​Ck​Ck​​=i∣j=k∑​Ai​Bj​=i&j=k∑​Ai​Bj​=i⊕j=k∑​Ai​Bj​​

FFT/NTT 就不能做了，需要用到 FMT/FWT。但是它们的想法是共通的：用另外的多项式来表示 AAA、BBB，然后再对应相乘，最后再变换回来。

根据准确的定义，快速莫比乌斯变换只能解决与和或的运算，快速沃尔什变换只能解决异或的运算。

快速莫比乌斯变换

或运算

我们构造 FMT[A]i=∑j∣i=iAjFMT[A]_i=\sum_{j|i=i}A_jFMT[A]i​=∑j∣i=i​Aj​，即 iii 对应二进制表示的所有子集的和。（类似高维前缀和的思想）
则有

FMT[A]×FMT[B]=(∑j∣i=iAj)(∑k∣i=iBk)=∑j∣i=i∑k∣i=iAjBk=∑(j∣k)∣i=iAjBk=FMT[C]\begin{aligned} FMT[A] \times FMT[B]&=\left(\sum_{j|i=i}A_j\right)\left(\sum_{k|i=i}B_k\right)\\ &=\sum_{j|i=i}\sum_{k|i=i}A_jB_k\\ &=\sum_{(j|k)|i=i}A_jB_k\\ &=FMT[C] \end{aligned} FMT[A]×FMT[B]​=⎝⎛​j∣i=i∑​Aj​⎠⎞​⎝⎛​k∣i=i∑​Bk​⎠⎞​=j∣i=i∑​k∣i=i∑​Aj​Bk​=(j∣k)∣i=i∑​Aj​Bk​=FMT[C]​

我们令 A0A_0A0​ 表示 AAA 的前一半，A1A_1A1​ 表示后一半，那么 A0A_0A0​ 就是 AAA 下标最高位为 000 的位，他的子集就是他本身的子集（因为最高位为 000 了），但是 A1A_1A1​ 的最高位是 111，他满足条件的子集不仅仅是他本身，还包最高位为 000 的子集，即

FMT[A]=merge⁡(FMT[A0],FMT[A0]+FMT[A1])FMT[A]=\operatorname{merge}(FMT[A_0],FMT[A_0]+FMT[A_1]) FMT[A]=merge(FMT[A0​],FMT[A0​]+FMT[A1​])

其中 merge⁡\operatorname{merge}merge 表示直接拼接。
那么 IFMT 也就很好实现

A=merge⁡(A0,A1−A0)A=\operatorname{merge}(A_0,A_1-A_0) A=merge(A0​,A1​−A0​)

与运算

和或运算类似，定义 FMT[A]=∑j&i=iAjFMT[A]=\sum_{j \& i=i}A_jFMT[A]=∑j&i=i​Aj​，有

FMT[A]=merge⁡(FMT[A0]+FMT[A1],FMT[A1])FMT[A]=\operatorname{merge}(FMT[A_0]+FMT[A_1],FMT[A_1]) FMT[A]=merge(FMT[A0​]+FMT[A1​],FMT[A1​])

A=merge⁡(A0−A1,A1)A=\operatorname{merge}(A_0-A_1,A_1) A=merge(A0​−A1​,A1​)

快速沃尔什变换

定义 x⊗y=popcount⁡(x&y) mod 2x \otimes y=\operatorname{popcount}(x \& y) \bmod 2x⊗y=popcount(x&y)mod2，其中 popcount⁡\operatorname{popcount}popcount 表示二进制下 111 的个数。
满足 (i⊗j)⊕(i⊗k)=i⊗(j⊕k)(i \otimes j) \oplus (i \otimes k)=i \otimes (j \oplus k)(i⊗j)⊕(i⊗k)=i⊗(j⊕k)。
构造 FWT[A]i=∑i⊗j=0Aj−∑i⊗j=1AjFWT[A]_i=\sum_{i \otimes j=0}A_j-\sum_{i \otimes j=1}A_jFWT[A]i​=∑i⊗j=0​Aj​−∑i⊗j=1​Aj​。
则有

FWT[A]×FWT[B]=(∑j⊗i=0Aj−∑j⊗i=1Aj)(∑k⊗i=0Bk−∑k⊗i=1Bk)=(∑j⊗i=0Aj)(∑k⊗i=0Bk)−(∑j⊗i=0Aj)(∑k⊗i=1Bk)−(∑j⊗i=1Aj)(∑k⊗i=0Bk)+(∑j⊗i=1Aj)(∑k⊗i=1Bk)=∑i⊗(j⊕k)=0AjBk−∑i⊗(j⊕k)=1AjBk=FWT[C]\begin{aligned} FWT[A] \times FWT[B]&=\left(\sum_{j \otimes i=0}A_j-\sum_{j \otimes i=1}A_j\right)\left(\sum_{k \otimes i=0}B_k-\sum_{k \otimes i=1}B_k\right)\\ &=\left(\sum_{j \otimes i=0}A_j\right)\left(\sum_{k \otimes i=0}B_k\right)-\left(\sum_{j \otimes i=0}A_j\right)\left(\sum_{k \otimes i=1}B_k\right)-\left(\sum_{j \otimes i=1}A_j\right)\left(\sum_{k \otimes i=0}B_k\right)+\left(\sum_{j \otimes i=1}A_j\right)\left(\sum_{k \otimes i=1}B_k\right)\\ &=\sum_{i \otimes (j \oplus k)=0}A_jB_k-\sum_{i \otimes (j \oplus k)=1}A_jB_k\\ &=FWT[C] \end{aligned} FWT[A]×FWT[B]​=(j⊗i=0∑​Aj​−j⊗i=1∑​Aj​)(k⊗i=0∑​Bk​−k⊗i=1∑​Bk​)=(j⊗i=0∑​Aj​)(k⊗i=0∑​Bk​)−(j⊗i=0∑​Aj​)(k⊗i=1∑​Bk​)−(j⊗i=1∑​Aj​)(k⊗i=0∑​Bk​)+(j⊗i=1∑​Aj​)(k⊗i=1∑​Bk​)=i⊗(j⊕k)=0∑​Aj​Bk​−i⊗(j⊕k)=1∑​Aj​Bk​=FWT[C]​

因此

FWT[A]=merge⁡(FWT[A0]+FWT[A1],FWT[A0]−FWT[A1])FWT[A]=\operatorname{merge}(FWT[A_0]+FWT[A_1],FWT[A_0]-FWT[A_1]) FWT[A]=merge(FWT[A0​]+FWT[A1​],FWT[A0​]−FWT[A1​])

A=merge⁡(A0+A12,A0−A12)A=\operatorname{merge}\left(\frac{A_0+A_1}{2},\frac{A_0-A_1}{2}\right) A=merge(2A0​+A1​​,2A0​−A1​​)

代码实现

#include <iostream>
#include <iomanip>
#include <cstdio>
#include <algorithm>
#include <cstring>
constexpr int kMod = 998244353;
int n, m, A[131072], B[131072], a[131072], b[131072];
void FMT_Or(int *a, int n, int op)
{
    for (int mid = 1; mid < n; mid <<= 1)
    {
        for (int i = 0; i < n; i += mid << 1)
        {
            for (int j = 0; j < mid; j++)
            {
                a[i + j + mid] = ((a[i + j + mid] + a[i + j] * op) % kMod + kMod) % kMod;
            }
        }
    }
}
void FMT_And(int *a, int n, int op)
{
    for (int mid = 1; mid < n; mid <<= 1)
    {
        for (int i = 0; i < n; i += mid << 1)
        {
            for (int j = 0; j < mid; j++)
            {
                a[i + j] = ((a[i + j] + a[i + j + mid] * op) % kMod + kMod) % kMod;
            }
        }
    }
}
void FWT_Xor(int *a, int n, int op)
{
    for (int mid = 1; mid < n; mid <<= 1)
    {
        for (int i = 0; i < n; i += mid << 1)
        {
            for (int j = 0; j < mid; j++)
            {
                int tmp1 = (a[i + j] + a[i + j + mid]) % kMod, tmp2 = (a[i + j] - a[i + j + mid] + kMod) % kMod;
                a[i + j] = tmp1 * static_cast<long long>(op) % kMod;
                a[i + j + mid] = tmp2 * static_cast<long long>(op) % kMod;
            }
        }
    }
}
int main(int argc, char const *argv[])
{
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::cin >> m;
    n = 1 << m;
    for (int i = 0; i < n; i++)
    {
        std::cin >> A[i];
    }
    for (int i = 0; i < n; i++)
    {
        std::cin >> B[i];
    }
    memcpy(a, A, 4 * n);
    memcpy(b, B, 4 * n);
    FMT_Or(a, n, 1);
    FMT_Or(b, n, 1);
    for (int i = 0; i < n; i++)
    {
        a[i] = a[i] * static_cast<long long>(b[i]) % kMod;
    }
    FMT_Or(a, n, -1);
    for (int i = 0; i < n; i++)
    {
        std::cout << a[i] << ' ';
    }
    std::cout << '\n';
    memcpy(a, A, 4 * n);
    memcpy(b, B, 4 * n);
    FMT_And(a, n, 1);
    FMT_And(b, n, 1);
    for (int i = 0; i < n; i++)
    {
        a[i] = a[i] * static_cast<long long>(b[i]) % kMod;
    }
    FMT_And(a, n, -1);
    for (int i = 0; i < n; i++)
    {
        std::cout << a[i] << ' ';
    }
    std::cout << '\n';
    memcpy(a, A, 4 * n);
    memcpy(b, B, 4 * n);
    FWT_Xor(a, n, 1);
    FWT_Xor(b, n, 1);
    for (int i = 0; i < n; i++)
    {
        a[i] = a[i] * static_cast<long long>(b[i]) % kMod;
    }
    FWT_Xor(a, n, 499122177);
    for (int i = 0; i < n; i++)
    {
        std::cout << a[i] << ' ';
    }
    std::cout << '\n';
    return 0;
}
